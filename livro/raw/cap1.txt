$h3 Calculadora para uma festinha

@py
def calcule_total(núm_pizzas, núm_refris):
    valor_pizza = 50
    valor_refri = 15

    return (núm_pizzas * valor_pizza) + (núm_refris * valor_refri)
@endpy

@py
aprox_π = 3.14
